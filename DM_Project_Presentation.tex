% ---- MOTIVATION
\begin{frame}{Motivation}
\begin{columns}[T]
  \column{0.6\textwidth}
  \begin{itemize}
    \item Air pollution is one of the most significant urban health challenges.
    \item Torino hosts both \textbf{Traffic} and \textbf{Background} monitoring stations,
          enabling direct pollutant comparison across station types.
    \item The project builds a \textbf{reproducible data-management pipeline}, not just
          a set of static charts.
    \item Analytical focus: PM10, NO2, exceedance days, integration quality.
  \end{itemize}
  \column{0.4\textwidth}
  \begin{tcolorbox}[kpi]
    {\tiny\bfseries\color{mutedgray}PIPELINE STAGES}\\[4pt]
    {\footnotesize
      1. Acquisition\\
      2. Storage (BigQuery)\\
      3. Transformation (dbt)\\
      4. Integration \& Quality\\
      5. Analysis \& Dashboard%
    }
  \end{tcolorbox}
\end{columns}
\end{frame}

% ---- RESEARCH QUESTIONS
\begin{frame}{Research Questions}
\begin{tcolorbox}[rqbox,title=RQ1 — Descriptive]
  How do PM10 and NO2 annual concentrations differ between Traffic and Background stations in 2022--2024?
\end{tcolorbox}
\vspace{4pt}
\begin{tcolorbox}[rqbox,title=RQ2 — Temporal]
  How do annual PM10 and NO2 indicators evolve between 2022 and 2024 for different station types?
\end{tcolorbox}
\vspace{4pt}
\begin{tcolorbox}[rqbox,title=RQ3 — Exceedance analysis]
  How do PM10 exceedance days above 50\,\textmu g/m\textsuperscript{3} vary across years and station types?
\end{tcolorbox}
\vspace{4pt}
\begin{tcolorbox}[rqbox,title=RQ4 — Integration and enrichment quality]
  Can yearly weather indicators and reverse-geocoded station information be integrated with measurable quality?
\end{tcolorbox}
\end{frame}

\textbf{Key findings:}

  \item PM10 annual means decrease over 2022--2024 for both station types.
  \item PM10 exceedance days exceed the EU limit at Traffic stations every year.
